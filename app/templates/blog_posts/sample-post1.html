# build the dataset and then split it into 80/10/10 for train/dev/test
block_size = 3 # context length: how many characters do we take to predict the next one?

def build_dataset(words):  
    X, Y = [], []
    for w in words:
        #print(w)
        context = [0] * block_size
        for ch in w + '.':
            ix = stoi[ch]
            X.append(context)
            Y.append(ix)
            #print(''.join(itos[i] for i in context), '--->', itos[ix])
            context = context[1:] + [ix] # crop and append

    X = torch.tensor(X)
    Y = torch.tensor(Y)
    print(X.shape, Y.shape)
    return X, Y

import random
random.seed(42)
random.shuffle(words)
n1 = int(0.8*len(words))
n2 = int(0.9*len(words))

Xtr, Ytr = build_dataset(words[:n1])
Xdev, Ydev = build_dataset(words[n1:n2])
Xte, Yte = build_dataset(words[n2:])

#Initialize the network - 1 tanh layer and then logits layer
tanh_layer_size = 200
embedding_dim = 10
g = torch.Generator().manual_seed(2147483647) # for reproducibility
C = torch.randn((total_i_o, embedding_dim), generator=g)
W1 = torch.randn((block_size*embedding_dim, tanh_layer_size), generator=g)
b1 = torch.randn(tanh_layer_size, generator=g)
W2 = torch.randn((tanh_layer_size, total_i_o), generator=g)
b2 = torch.randn(total_i_o, generator=g)
parameters = [C, W1, b1, W2, b2]
#Need to let Torch know that all our parameters will be backpropagated
#Therefore we "turn on" the logging/tracking of each parameter's gradient (i.e derivative)
for p in parameters:
    p.requires_grad = True
    
#For tracking our loss relative to the step number because we're using batching
lossi = []
stepi = []
batch_size = 3200
#Gradient descent for optimizing the network
for i in range(200000):
    
    # minibatch construct
    ix = torch.randint(0, Xtr.shape[0], (batch_size,))
    
    # forward pass using batch
    emb = C[Xtr[ix]] # (32, 3, 10)
    h = torch.tanh(emb.view(-1, block_size*embedding_dim) @ W1 + b1) #-1 lets torch infer that dimm based on dimm of W1
    logits = h @ W2 + b2 # (32, 27)
    loss = F.cross_entropy(logits, Ytr[ix])
    #print(loss.item())
  
    # backward pass
    for p in parameters:
        p.grad = None #must reset the grad before going backwards or it accumulates
    loss.backward()
  
    # update with a basic learning rate
    lr = 0.1 if i < 100000 else 0.01
    for p in parameters:
        p.data += -lr * p.grad

    # track stats
    stepi.append(i)
    lossi.append(loss.log10().item())
#Obtain the loss from the Dev/Vald set
emb = C[Xtr] # (32, 3, 2)
h = torch.tanh(emb.view(-1, block_size*embedding_dim) @ W1 + b1) # (32, 100)
logits = h @ W2 + b2 # (32, 27)
loss = F.cross_entropy(logits, Ytr)

        logits = h @ W2 + b2
        probs = F.softmax(logits, dim=1)
        ix = torch.multinomial(probs, num_samples=1, generator=g).item()
        context = context[1:] + [ix]

        if ix == 0:
            break
